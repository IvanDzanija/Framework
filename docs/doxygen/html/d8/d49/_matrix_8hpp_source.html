416 if (_cols != other.row_count()) {

424 const size_t a_cols = _cols;

428 const U *b_data = other.data().data();
429 R *c_data = result.data().data();

431 result.fill(R(0));

433#pragma omp parallel for collapse(2) if (a_rows * b_cols > 10000)
434 for (size_t ii = 0; ii < a_rows; ii += BLOCK_SIZE) {
435 for (size_t jj = 0; jj < b_cols; jj += BLOCK_SIZE) {
436 for (size_t kk = 0; kk < a_cols; kk += BLOCK_SIZE) {
437 // Process block
438 const size_t i_end = std::min(ii + BLOCK_SIZE, a_rows);
439 const size_t j_end = std::min(jj + BLOCK_SIZE, b_cols);
440 const size_t k_end = std::min(kk + BLOCK_SIZE, a_cols);

499 [[nodiscard]] auto operator*(const Vector<U> &other) const {
500 using R = std::common_type_t<T, U>;
501
502 const size_t n = this->row_count();
503 const size_t m = this->column_count();
504
505 if (other.orientation() == Vector<U>::ROW) {
506 throw std::invalid_argument(
507 "Invalid multiplication: matrix * row vector.\n"
508 "Did you mean Vector * Matrix?");
509 }
510
511 if (other.size() != m) {
512 throw std::invalid_argument(
513 "Dimension mismatch in Matrix * Vector multiplication.");
514 }
515
516 Vector<R> result(n, std::vector<R>(n, R(0)), Vector<R>::COLUMN);
517
518#pragma omp parallel for
519 for (size_t i = 0; i < n; ++i) {
520 R sum = R(0);
521 auto L_row_i = this->row_span(i);
522#pragma omp simd reduction(+ : sum)
523 for (size_t j = 0; j < m; ++j) {
524 sum += static_cast<R>(L_row_i[j]) * static_cast<R>(other.at(j));
525 }
526 result.at(i) = sum;
527 }
528 return result;
